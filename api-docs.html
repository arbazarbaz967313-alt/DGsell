Code Examples
JAVASCRIPT
// Remove background using BG Eraser Pro API
const removeBg = async (imageFile) => {
  const form = new FormData();
  form.append('image', imageFile);

  const res = await fetch('https://bgeraserpro.com/api/v1/remove-bg', {
    method: 'POST',
    headers: { 'X-API-Key': 'YOUR_KEY_HERE' },
    body: form
  });
  return await res.json(); // { image_url, credits_remaining }
};
PYTHON
import requests

def remove_bg(path):
    with open(path, 'rb') as f:
        r = requests.post(
            'https://bgeraserpro.com/api/v1/remove-bg',
            headers={'X-API-Key': 'YOUR_KEY_HERE'},
            files={'image': f}
        )
    return r.json() # { image_url, credits_remaining }
