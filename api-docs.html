Code Example
JAVASCRIPT / FETCH 
// Send image to BG Eraser Pro API
const removeBackground = async (imageFile) => {
  const formData = new FormData();
  formData.append('image', imageFile);

  const response = await fetch('https://bgeraserpro.com/api/v1/remove-bg', {
    method: 'POST',
    headers: {
      'X-API-Key': 'YOUR_API_KEY_HERE'
    },
    body: formData
  });

  const data = await response.json();
  // data.image_url → transparent PNG download link
  // data.credits_used → 1
  // data.credits_remaining → your balance
  return data.image_url;
};
PYTHON 
import requests

def remove_background(image_path):
    with open(image_path, 'rb') as f:
        response = requests.post(
            'https://bgeraserpro.com/api/v1/remove-bg',
            headers={'X-API-Key': 'YOUR_API_KEY_HERE'},
            files={'image': f}
        )
    return response.json()
